library IEEE;
use IEEE.STD_LOGIC_1164.ALL;
use IEEE.NUMERIC_STD.ALL;

entity traffic_light is
    Port (
        clk   : in  STD_LOGIC;   
        reset : in  STD_LOGIC;   
        red   : out STD_LOGIC;
        yellow: out STD_LOGIC;
        green : out STD_LOGIC
    );
end traffic_light;

architecture tlc_arch of traffic_light is

    
    type state_type is (S_RED, S_YELLOW, S_GREEN);
    signal state : state_type := S_RED;

    -- Clock divider
    signal counter : unsigned(25 downto 0) := (others => '0'); 
    constant MAX_COUNT : unsigned(25 downto 0) := to_unsigned(50_000_000, 26); 
    

begin

    
    -- Clock Divider (1 second tick)
   
    process(clk, reset)
    begin
        if reset = '1' then
            counter <= (others => '0');
        elsif rising_edge(clk) then
            if counter = MAX_COUNT then
                counter <= (others => '0');
            else
                counter <= counter + 1;
            end if;
        end if;
    end process;

   
    -- State Machine
   
    process(clk, reset)
    begin
        if reset = '1' then
            state <= S_RED;
        elsif rising_edge(clk) then
            if counter = MAX_COUNT then
                case state is
                    when S_RED    => state <= S_GREEN;
                    when S_GREEN  => state <= S_YELLOW;
                    when S_YELLOW => state <= S_RED;
                end case;
            end if;
        end if;
    end process;

   
    -- Output Logic
   
    red    <= '1' when state = S_RED else '0';
    yellow <= '1' when state = S_YELLOW else '0';
    green  <= '1' when state = S_GREEN else '0';

end tlc_arch;
